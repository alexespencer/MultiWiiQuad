(function()
{
	"use strict"
	var bingNamespace = window.Bing = window.Bing || {};
	var bingMapNamespace = bingNamespace.MapControl = bingNamespace.MapControl || {};
	var C_maxTiles = 16;
	var C_maxAttempts = 64;
	var Direction =
		{
			right: 0,
			up: 1,
			left: 2,
			down: 3
		};

	// Tile primer preloads tiles by requesting them while the map control scripts are loading
	// - parentDiv - The div where the map will reside and where the preload tiles will reside
	// - zoom - The zoom level for the initial view of the map
	// - centerTileX - The x coordinate of the center tile (in map tile space from 0 to 2^zoom -1)
	// - centerTileY - The y coordinate of the center tile (in map tile space from 0 to 2^zoom -1)
	// - centerOffset - The offset to position the center tile so that the desired lat/long is in the center
	// - tileFormatKey - The key to the value in Microsoft.Maps.Globals with the correct tile format
	bingMapNamespace.TilePrimer = function(parentDiv, zoom, centerTileX, centerTileY, centerOffset, tileFormatKey)
	{
		var self = this;
		var Globals = Microsoft.Maps.Globals;
		var _tileFormat = Globals[tileFormatKey];
		var _width = parentDiv.clientWidth;
		var _height = parentDiv.clientHeight;
		var _maxTileCoordinate = Math.pow(2, zoom) - 1;
		// centerOffset is the offset from the center point where the center tile should be placed (always negative)
		// adjust it to be relative to the container's top left
		centerOffset.x += _width / 2;
		centerOffset.y += _height / 2;
		var _tileSize = Globals.defaultTileSize;

		// Couldn't determine which tiles to show, skip primer
		if (!_tileFormat) 
		{
			return;
		}

		var _tileContainer = document.createElement("div");
		_tileContainer.style.cssText = "position:relative;overflow:hidden;width:" + _width + "px;height:" + _height + "px";
		parentDiv.appendChild(_tileContainer);

		//---------------------------------------------------------
		// Spiral position variables
		//---------------------------------------------------------
		// The x coordinate of the current tile from 0 - 2^zoom - 1
		var _tileX;
		// The y coordinate of the current tile from 0 - 2^zoom - 1
		var _tileY;
		// The offset in pixels from the top-left of the container to place the tile
		var _offset;
		// The current direction of the spiral
		var _direction;
		// The number of moves remaining before turning
		var _movesLeft;
		// The current step in the progress of spiraling  the series goes 1, 1.5, 2, 2.5, 3, 3.5,... but only the
		// integer part is used effectivly giving a series of 1,1,2,2,3,3,4,4 which is the length of the segments
		// when spiraling
		var _step;

		// Add the tiles after page load because the tile loading will prevent the page load event
		addTiles();

		// Adds a single tile to the the UI to request the tile image
		function addTile(quadKey, offset)
		{
			var img = document.createElement("img");
			img.src = _tileFormat.replace("{quadkey}", quadKey).replace("{subdomain}", quadKey.charAt(quadKey.length - 1));
			img.src.replace("http:", window.location.protocol);
			img.style.cssText = "position:absolute;top:" + offset.y + "px;left:" + offset.x + "px";
			_tileContainer.appendChild(img);
		}

		// Adds all visible tiles to request their images
		function addTiles()
		{
			// Initialize the spiraling state
			_tileX = centerTileX;
			_tileY = centerTileY;
			_offset = centerOffset;
			_direction = Direction.right;
			_movesLeft = 1;
			_step = 1;

			// Attempt to add up to C_maxAttempts times or until C_maxTiles were successfully added
			for (var tileCount = 0, attemptCount = 0; attemptCount < C_maxAttempts && tileCount < C_maxTiles; attemptCount++)
			{
				if (isVisible(_offset))
				{
					var quadKey = getQuadKey();
					if (quadKey)
					{
						addTile(quadKey, _offset);
						tileCount++;
					}
				}
				incrementSpiral();
			}
		}

		// Gets the quad key for the current tileX and tileY
		function getQuadKey()
		{
			if (_tileY < 0 || _tileY > _maxTileCoordinate)
			{
				// We don't wrap around the north/south edge, so this is an invalid tile
				return null;
			}
			var quadKey = '';
			for (var i = zoom; i > 0; i--)
			{
				var digit = 0;
				var mask = 1 << (i - 1);
				if ((_tileX & mask) != 0)
				{
					digit++;
				}
				if ((_tileY & mask) != 0)
				{
					digit+=2;
				}
				quadKey += digit;
			}
			return quadKey;
		}

		// Determines if a tile offset would make any part of the tile visible
		function isVisible(offset)
		{
			return ((offset.y > -_tileSize) &&
					(offset.y < _height) &&
					(offset.x > -_tileSize) &&
					(offset.x < _width));
		}

		// Moves around the spiral
		function incrementSpiral()
		{
			switch (_direction)
			{
				case Direction.right:
					_tileX++;
					_offset.x += _tileSize;
					break;
				case Direction.up:
					_tileY--;
					_offset.y -= _tileSize;
					break;
				case Direction.left:
					_tileX--;
					_offset.x -= _tileSize;
					break;
				case Direction.down:
					_tileY++;
					_offset.y += _tileSize;
					break;
			}

			// Perform any wrap-around if necessary
			if (_tileX > _maxTileCoordinate)
			{
				_tileX = 0;
			}
			else if (_tileX < 0)
			{
				_tileX = _maxTileCoordinate;
			}

			_movesLeft--;
			if (_movesLeft <= 0)
			{
				// Increment a half step and reset movesLeft
				_movesLeft = Math.floor(_step += 0.5);
				// Rotate the direction and wrap if needed
				if (++_direction > Direction.down)
				{
					_direction = Direction.right;
				}
			}
		}
	};
